87 void
88 SSLHTTPDownloader::do_session(
89  std::string const host,
90  std::string const port,
91  std::string const target,
92  int version,
93  boost::filesystem::path dstPath,
94  std::function<void(boost::filesystem::path)> complete,
95  boost::asio::yield_context yield)
96 {
97  using namespace boost::asio;
98  using namespace boost::beast;
99 
100  boost::system::error_code ec;
101  ip::tcp::resolver resolver {strand_.context()};
102  auto const results = resolver.async_resolve(host, port, yield[ec]);
103  if (ec)
104  return fail(dstPath, complete, ec, "async_resolve");
105 
106  try
107  {
108  stream_.emplace(strand_.context(), ssl_ctx_.context());
109  }
110  catch (std::exception const& e)
111  {
112  return fail(dstPath, complete, ec,
113  std::string("exception: ") + e.what());
114  }
115 
116  ec = ssl_ctx_.preConnectVerify(*stream_, host);
117  if (ec)
118  return fail(dstPath, complete, ec, "preConnectVerify");
119 
120  boost::asio::async_connect(
121  stream_->next_layer(), results.begin(), results.end(), yield[ec]);
122  if (ec)
123  return fail(dstPath, complete, ec, "async_connect");
124 
125  ec = ssl_ctx_.postConnectVerify(*stream_, host);
126  if (ec)
127  return fail(dstPath, complete, ec, "postConnectVerify");
128 
129  stream_->async_handshake(ssl::stream_base::client, yield[ec]);
130  if (ec)
131  return fail(dstPath, complete, ec, "async_handshake");
132 
133  // Set up an HTTP HEAD request message to find the file size
134  http::request<http::empty_body> req {http::verb::head, target, version};
135  req.set(http::field::host, host);
136  req.set(http::field::user_agent, BOOST_BEAST_VERSION_STRING);
137 
138  http::async_write(*stream_, req, yield[ec]);
139  if(ec)
140  return fail(dstPath, complete, ec, "async_write");
141 
142  {
143  // Check if available storage for file size
144  http::response_parser<http::empty_body> p;
145  p.skip(true);
146  http::async_read(*stream_, read_buf_, p, yield[ec]);
147  if(ec)
148  return fail(dstPath, complete, ec, "async_read");
149  if (auto len = p.content_length())
150  {
151  try
152  {
153  if (*len > space(dstPath.parent_path()).available)
154  {
155  return fail(dstPath, complete, ec,
156  "Insufficient disk space for download");
157  }
158  }
159  catch (std::exception const& e)
160  {
161  return fail(dstPath, complete, ec,
162  std::string("exception: ") + e.what());
163  }
164  }
165  }
166 
167  // Set up an HTTP GET request message to download the file
168  req.method(http::verb::get);
169  http::async_write(*stream_, req, yield[ec]);
170  if(ec)
171  return fail(dstPath, complete, ec, "async_write");
172 
173  // Download the file
174  http::response_parser<http::file_body> p;
175  p.body_limit(std::numeric_limits<std::uint64_t>::max());
176  p.get().body().open(
177  dstPath.string().c_str(),
178  boost::beast::file_mode::write,
179  ec);
180  if (ec)
181  {
182  p.get().body().close();
183  return fail(dstPath, complete, ec, "open");
184  }
185 
186  http::async_read(*stream_, read_buf_, p, yield[ec]);
187  if (ec)
188  {
189  p.get().body().close();
190  return fail(dstPath, complete, ec, "async_read");
191  }
192  p.get().body().close();
193 
194  // Gracefully close the stream
195  stream_->async_shutdown(yield[ec]);
196  if (ec == boost::asio::error::eof)
197  ec.assign(0, ec.category());
198  if (ec)
199  {
200  // Most web servers don't bother with performing
201  // the SSL shutdown handshake, for speed.
202  JLOG(j_.trace()) <<
203  "async_shutdown: " << ec.message();
204  }
205  // The socket cannot be reused
206  stream_ = boost::none;
207 
208  JLOG(j_.trace()) <<
209  "download completed: " << dstPath.string();
210 
211  // Notify the completion handler
212  complete(std::move(dstPath));
213 }

215 void
216 SSLHTTPDownloader::fail(
217  boost::filesystem::path dstPath,
218  std::function<void(boost::filesystem::path)> const& complete,
219  boost::system::error_code const& ec,
220  std::string const& errMsg)
221 {

229  JLOG(j_.error()) <<
230  errMsg << ": " << ec.message();
231  }
232 
233  try

235  remove(dstPath);
236  }
237  catch (std::exception const& e)
238  {
239  JLOG(j_.error()) <<
240  "exception: " << e.what();
241  }
242  complete(std::move(dstPath));
243 }
