120 template <class T>
121 boost::optional<T>
122 prevMissing(RangeSet<T> const & rs, T t, T minVal = 0)
123 {
124  if (rs.empty() || t == minVal)
125  return boost::none;
126  RangeSet<T> tgt{ ClosedInterval<T>{minVal, t - 1} };
127  tgt -= rs;
128  if (tgt.empty())
129  return boost::none;
130  return boost::icl::last(tgt);
131 }
