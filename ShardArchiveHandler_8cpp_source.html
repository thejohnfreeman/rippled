67 ShardArchiveHandler::add(std::uint32_t shardIndex, parsedURL&& url)
68 {
69  std::lock_guard lock(m_);
70  if (process_)
71  {
72  JLOG(j_.error()) <<
73  "Download and import already in progress";
74  return false;
75  }
76 
77  auto const it {archives_.find(shardIndex)};
78  if (it != archives_.end())
79  return url == it->second;
80  if (!app_.getShardStore()->prepareShard(shardIndex))
81  return false;
82  archives_.emplace(shardIndex, std::move(url));
83  return true;

86 bool
87 ShardArchiveHandler::start()
88 {
89  std::lock_guard lock(m_);
90  if (!app_.getShardStore())
91  {
92  JLOG(j_.error()) <<
93  "No shard store available";
94  return false;
95  }
96  if (process_)
97  {
98  JLOG(j_.warn()) <<
99  "Archives already being processed";
100  return false;
101  }
102  if (archives_.empty())
103  {
104  JLOG(j_.warn()) <<
105  "No archives to process";
106  return false;
107  }
108 
109  try
110  {
111  // Remove if remnant from a crash
112  remove_all(downloadDir_);
113 
114  // Create temp root download directory
115  create_directory(downloadDir_);
116 
117  if (!downloader_)
118  {
119  // will throw if can't initialize ssl context
120  downloader_ = std::make_shared<SSLHTTPDownloader>(
121  app_.getIOService(), j_, app_.config());
122  }
123  }
124  catch (std::exception const& e)
125  {
126  JLOG(j_.error()) <<
127  "exception: " << e.what();
128  return false;
129  }
130 
131  return next(lock);
132 }

134 bool
135 ShardArchiveHandler::next(std::lock_guard<std::mutex>& l)
136 {
137  if (archives_.empty())
138  {
139  process_ = false;
140  return false;
141  }
142 
143  // Create a temp archive directory at the root
144  auto const shardIndex {archives_.begin()->first};
145  auto const dstDir {downloadDir_ / std::to_string(shardIndex)};
146  try
147  {
148  create_directory(dstDir);
149  }
150  catch (std::exception const& e)
151  {
152  JLOG(j_.error()) <<
153  "exception: " << e.what();
154  remove(l);
155  return next(l);
156  }
157 
158  // Download the archive
159  auto const& url {archives_.begin()->second};
160  if (!downloader_->download(
161  url.domain,
162  std::to_string(url.port.get_value_or(443)),
163  url.path,
164  11,
165  dstDir / "archive.tar.lz4",
166  std::bind(&ShardArchiveHandler::complete,
167  shared_from_this(), std::placeholders::_1)))
168  {
169  remove(l);
170  return next(l);
171  }
172 
173  process_ = true;
174  return true;
175 }

177 void
178 ShardArchiveHandler::complete(path dstPath)
179 {
180  {
181  std::lock_guard lock(m_);
182  try
183  {
184  if (!is_regular_file(dstPath))
185  {
186  auto ar {archives_.begin()};
187  JLOG(j_.error()) <<
188  "Downloading shard id " << ar->first <<
189  " form URL " << ar->second.domain << ar->second.path;
190  remove(lock);
191  next(lock);
192  return;
193  }
194  }
195  catch (std::exception const& e)
196  {
197  JLOG(j_.error()) <<
198  "exception: " << e.what();
199  remove(lock);
200  next(lock);
201  return;
202  }
203  }
204 
205  // Process in another thread to not hold up the IO service
206  app_.getJobQueue().addJob(
207  jtCLIENT, "ShardArchiveHandler",
208  [=, dstPath = std::move(dstPath), ptr = shared_from_this()](Job&)
209  {
210  // If validating and not synced then defer and retry
211  auto const mode {ptr->app_.getOPs().getOperatingMode()};
212  if (ptr->validate_ && mode != OperatingMode::FULL)
213  {
214  std::lock_guard lock(m_);
215  timer_.expires_from_now(static_cast<std::chrono::seconds>(
216  (static_cast<std::size_t>(OperatingMode::FULL)
217  - static_cast<std::size_t>(mode)) * 10));
218  timer_.async_wait(
219  [=, dstPath = std::move(dstPath), ptr = std::move(ptr)]
220  (boost::system::error_code const& ec)
221  {
222  if (ec != boost::asio::error::operation_aborted)
223  ptr->complete(std::move(dstPath));
224  });
225  }
226  else
227  {
228  ptr->process(dstPath);
229  std::lock_guard lock(m_);
230  remove(lock);
231  next(lock);
232  }
233  });
234 }

236 void
237 ShardArchiveHandler::process(path const& dstPath)
238 {
239  std::uint32_t shardIndex;
240  {
241  std::lock_guard lock(m_);
242  shardIndex = archives_.begin()->first;
243  }
244 
245  auto const shardDir {dstPath.parent_path() / std::to_string(shardIndex)};
246  try
247  {
248  // Extract the downloaded archive
249  extractTarLz4(dstPath, dstPath.parent_path());
250 
251  // The extracted root directory name must match the shard index
252  if (!is_directory(shardDir))
253  {
254  JLOG(j_.error()) <<
255  "Shard " << shardIndex <<
256  " mismatches archive shard directory";
257  return;
258  }
259  }
260  catch (std::exception const& e)
261  {
262  JLOG(j_.error()) <<
263  "exception: " << e.what();
264  return;
265  }
266 
267  // Import the shard into the shard store
268  if (!app_.getShardStore()->importShard(shardIndex, shardDir, validate_))
269  {
270  JLOG(j_.error()) <<
271  "Importing shard " << shardIndex;
272  return;
273  }
274 
275  JLOG(j_.debug()) <<
276  "Shard " << shardIndex << " downloaded and imported";
277 }
