30  void
31  testPrevMissing()
32  {
33  testcase("prevMissing");
34 
35  // Set will include:
36  // [ 0, 5]
37  // [10,15]
38  // [20,25]
39  // etc...
40 
41  RangeSet<std::uint32_t> set;
42  for (std::uint32_t i = 0; i < 10; ++i)
43  set.insert(range(10 * i, 10 * i + 5));
44 
45  for (std::uint32_t i = 1; i < 100; ++i)
46  {
47  boost::optional<std::uint32_t> expected;
48  // no prev missing in domain for i <= 6
49  if (i > 6)
50  {
51  std::uint32_t const oneBelowRange = (10 * (i / 10)) - 1;
52 
53  expected = ((i % 10) > 6) ? (i - 1) : oneBelowRange;
54  }
55  BEAST_EXPECT(prevMissing(set, i) == expected);
56  }
57  }
