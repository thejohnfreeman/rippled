283 bool
284 multi_runner_child::run_multi(Pred pred)
285 {
286  auto const& suite = beast::unit_test::global_suites();
287  auto const num_tests = suite.size();
288  bool failed = false;
289 
290  auto get_test = [&]() -> beast::unit_test::suite_info const* {
291  auto const cur_test_index = checkout_test_index();
292  if (cur_test_index >= num_tests)
293  return nullptr;
294  auto iter = suite.begin();
295  std::advance(iter, cur_test_index);
296  return &*iter;
297  };
298  while (auto t = get_test())
299  {
300  if (!pred(*t))
301  continue;
302  try
303  {
304  failed = run(*t) || failed;
305  }
306  catch (...)
307  {
308  if (num_jobs_ <= 1)
309  throw; // a single process can die
310 
311  // inform the parent
312  std::stringstream s;
313  s << job_index_ << "> failed Unhandled exception in test.\n";
314  message_queue_send(MessageType::log, s.str());
315  failed = true;
316  }
317  }
318  any_failed(failed);
319  return failed;
320 }
